# VoiceBee sidecar — direct dependencies (Phase 2, DECISIONS Q12).

# Kokoro-82M text processing (the fast interactive Talk voice). misaki is the G2P
# front-end; it needs num2words + spaCy (+ the en_core_web_sm model) for English,
# and phonemizer for out-of-dictionary words. cp314 wheels exist for all of these.
# NOTE: phonemizer needs the espeak-ng SYSTEM library at runtime (brew install
# espeak-ng on dev; must be bundled/installed for shipped DMG users — see #4c).
misaki==0.7.4
num2words==0.5.14
spacy==3.8.13
phonemizer==3.3.0
https://github.com/explosion/spacy-models/releases/download/en_core_web_sm-3.8.0/en_core_web_sm-3.8.0-py3-none-any.whl

# Realtime voice (P5 — the iOS WebRTC client). The [webrtc] extra adds aiortc +
# PyAV + pylibsrtp + opencv (P2P SmallWebRTC transport, no cloud SFU); [silero]
# adds the VAD used for realtime turn-taking + native barge-in. All ship arm64
# wheels for Python 3.14 (verified — no source builds). websockets is a base
# transitive dep not pulled by the narrow pin. NOTE: av and opencv each bundle
# their own ffmpeg dylibs (a harmless objc duplicate-class warning for our
# audio-only path) — dedupe when bundling into the signed runtime (#4c).
pipecat-ai[webrtc,silero]==1.3.0   # realtime pipeline: SmallWebRTC + VAD/STT/LLM/TTS
websockets==16.0
